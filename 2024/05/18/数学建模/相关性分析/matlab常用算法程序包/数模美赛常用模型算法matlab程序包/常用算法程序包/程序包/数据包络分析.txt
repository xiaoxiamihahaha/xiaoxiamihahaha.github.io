%输入指标数据与输出指标数据
X=[89.39 86.25 108.13 106.38 62.40 47.19;64.3 99 99.6 96 96.2 79.9];

Y=[25.2 28.2 29.4 26.4 27.2 25.2;223 287 317 291 295 222];

[m,s]=size(X);

[n,~]=size(Y);

b=ones(s,1);

A=[Y',-X'];

result=zeros(n,1);

for i=1:s
    
      f=[-Y(:,i)',-X(:,i)'];
    
      x=linprog(f,A,b,[],[],zeros(m+n,1));
    
      result(i,1)=(f(1,1:m)*x(1:m,:))/(f(:,m+1:end)*x(m+1:end,:));

end

result
